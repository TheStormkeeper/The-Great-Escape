A11D: play_speaker
Makes a sound through the speaker.
B Number of iterations to play for.
C Delay inbetween each iteration.
A11D LD A,C delay = C; // Self-modify delay loop at $A126.
A11E LD ($A126),A
A121 LD A,$10 A = 16; // Initial speaker bit.
A123 OUT ($FE),A do { OUT ($FE),A // Play. } {C = delay; while (C--) ; } A ^= 16; // Toggle speaker bit. while (--B);
A125 LD C,$37
A127 DEC C
A128 JR NZ,$A127
A12A XOR $10
A12C DJNZ $A123
A12E RET return;
